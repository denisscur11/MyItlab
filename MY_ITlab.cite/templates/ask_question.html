// Автоматична система підтримки
class SupportBot {
  constructor() {
    this.responseTime = "24 години";
    this.topics = ["Навчання", "Оплата", "Технічні питання"];
  }
  
  async processQuestion(question, email) {
    console.log(`Нове питання від ${email}`);
    
    // Відправляємо на пошту підтримки
    await this.notifySupport(question);
    
    return {
      status: "success",
      message: "Ваше питання отримано!"
    };
  }
}

// Запуск сервісу
const supportService = new SupportBot();
